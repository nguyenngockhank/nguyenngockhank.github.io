# struct

package main

import (
	"fmt"
)

type Person struct {
	Name string 
	Age int
}

// pass-by-reference
func (p *Person) Grow() { 
	p.Age++
}

// pass-by-value
func (p Person) NoGrow() {
	p.Age++
}


func main() {
	p := Person{"Khank", 10}
    p2 := p // clone 
    p3 := &p
    
    p.Grow()
	p2.Grow()
	p3.Grow() // shorthand for: (&p).Grow()
	
	fmt.Println(p.Age)  // 12
	fmt.Println(p2.Age) // 11
	fmt.Println(p3.Age) // 12
}

Pass-By-Reference:

  • want to modify the receiver
  • the struct is very large & a deep copy is expensive

# embedding

Embedding

Embedding is a mechanism to allow the ability to borrow pieces from different classes. It is the quivalent of multiple inheritance with non-virtual members.

type Bird struct {
    FeatherLength   int
    classification  string
}

type Pigeon struct {
    Bird
    Name string
}

func main() {
    p := Pigeon{ Name: "Tweety" }
    p.FeatherLength = 10
}
