# Loop

# For in Range

for (let i = 0; i < 5; i++) {
    console.log(val)  // 0, 1, 2, 3, 4
}

for x in range(0, 4):
  print(x) # 0, 1, 2, 3, 4

for ($i = 0; $i < 5; $i++) {
    // ...
}

for (int i = 0; i < 5; i++) {
  System.out.println(i);
}

for i := 0; i < 5; i++ {
    fmt.Println(i)
}

# For through an Array

Access elements of an array

let arr = [10, 20, 30, 40]

for (let val of arr) {
    console.log(val)  // 10, 20, 30, 40
}

arr = [10, 20, 30, 40]

for x in arr:
    print(x)  # 10, 20, 30, 40

$arr = [10, 20, 30, 40];

foreach($arr as $val) {
    echo $val; // 10, 20, 30, 40
}

int[] numbers = {10, 20, 30, 40};

for (int i : numbers) {
  System.out.println(i);
}

arr := []int{10, 20, 30, 40}

for k, val := range arr {
    fmt.Println(k, val)
}

# Loop Through a Dictionary

Access properties of an object

var person = {
    fname:"John", 
    lname:"Doe", 
    age:25
};

for (var x in person) {
    console.log(x); // fname, lname, age
    console.log(person[x]) // John, Doe, 25
}

person = {
  "fname": "John",
  "lname": "Doe",
  "age": 25
}
for k, v in person.items():
  print(k) # fname, lname, age
  print(v) # John, Doe, 25

$person = [
    "fname" => "John", 
    "lname" => "Doe", 
    "age" => 25
];

foreach($person as $key => $val) {
    echo $key; // fname, lname, age
    echo $val; // John, Doe, 25
}

Map<String, Object> map = new HashMap<String, Object>();

map.put("fname", "John");

map.put("age", 25);
    
for ( Map.Entry<String, Object> entry : map.entrySet() ) {
    String key = entry.getKey();
    Object value = entry.getValue();
    
    System.out.println(key + " " + value);

// another types

m := map[string]string{
    "fname": "John",
    "lname":  "Doe",
