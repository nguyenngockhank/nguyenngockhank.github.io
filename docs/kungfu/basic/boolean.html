# Boolean

# Definition

var f = false
var t = true

let f : boolean = false
let t : boolean = true

t = True
f = False

$t = true;
$f = false;

boolean t = true;
boolean f = false;

# Not

let t = true
let f = !t

let t : boolean = true
let f = !t

t = True
f = not t

$t = true;
$f = !$t;

boolean t = true;
boolean f = !t;
