# HTML Template

html-webpack-plugin (opens new window)

npm i -D html-webpack-plugin

webpack.*.js

const HtmlWebpackPlugin = require('html-webpack-plugin');

const InjectTemplates = [
    'index.html', 
].map(filename => new HtmlWebpackPlugin({
    filename,
    inject: false,
    template: path.join(__dirname, 'templates', filename),

    templateParameters: (compilation, assets, assetTags, options) => {

        return {
            compilation,
            webpackConfig: compilation.options,
            htmlWebpackPlugin: {
                tags: assetTags,
                files: assets,
                options
            },
            'name': 'khank' // variable will be passed to template
        };
    },
}))


let devOpts = merge(common, {
    plugins: [
        ...InjectTemplates
    ],
}
